/*
  제네릭 클래스 만드는 방법
  형식)
  class 클래스명<제네릭타입문자> {
  
  	제네릭타입문자 변수명; => 변수선언에 제네릭을 사용하는 경우
  	
  	//반환값이 있는 메소드에 제네릭을 사용하는 경우
  	제네릭타입문자 메소드명() { 
  		...
  		return 반환값;
  	}
  	
  	//메소드의 매개변수에 제네릭을 사용하는 경우
  	void 메소드명(제네틸가입문자 변수명, ...) {
  		...
  	}
  	
  }
  
  -- 제네릭 타입 글자 --
  T => Type
  K => Key
  V => Value
  E => Element (자료구조에 들어가는 데이터)
 */
